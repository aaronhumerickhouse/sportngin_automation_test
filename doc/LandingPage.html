def sign_in_with_google(username, password)
  self.google_sign_in.click
  sleep 2
  $browser.send_keys username
  $browser.send_keys :enter
  sleep 2
  $browser.send_keys password
  $browser.send_keys :enter
  return BoardsPage.new $browser
end
